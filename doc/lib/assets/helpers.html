Dependencies.

var fs = require('fs'); var pathM = require('path'); var utils = require('../utilities');

Resolves an asset-relative path to an absolute path.

Parameters:

  • root must be a String.
    (The application root.)

  • paths must be an Array.
    (An array containing asset paths.)

  • path must be a String.
    (The asset-relative path.)

  • callback must be a Function.
    (Called when finished.)

  • index is optional, must be a Number, and has a default value of 0.
    (Start at this asset path.)

module.exports.resolve = function resolve(root, paths, path, callback, index) { var fullpath; index = index || 0; if (index >= paths.length) { return callback(false); } fullpath = pathM.join(root, paths[index], path); fs.exists(fullpath, function(exists) { if (exists) { callback(true, fullpath); } else { resolve(root, paths, path, callback, index + 1); } }); };
