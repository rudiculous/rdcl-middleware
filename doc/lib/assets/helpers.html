Dependencies.

var fs = require('fs'); var glob = require('glob'); var pathM = require('path'); var utils = require('../utilities');

Resolves all globs (synchronously).

Since this method runs synchronously, it should only be called once during initialization. It should never run during requests!

Parameters:

  • root must be a String.
    (The globs are resolved relative to this path.)

  • paths must be an Array.
    (These paths are prepended to each file.)

  • type must be a String.
    (The asset type, which is prepended to each file.)

  • files must be an Object.
    (All asset files.)

Returns an Object

module.exports.resolveGlobs = function resolveGlobs(root, paths, type, files) { var file, assets, resolved, fullPath, result = {}; for (file in files) { resolved = []; assets = files[file]; assets.forEach(function(asset) { paths.forEach(function(path) { fullPath = pathM.join(path, type, asset); glob.sync(fullPath, { 'cwd': root, 'root': root, }).forEach(function(res) { resolved.push(res); }); }); }); result[file] = resolved; } return result; }
